\begin{sang}{Hvad skal vi kode i nat}{Lorenzen - Melodi - Hvor skal vi sove i nat}
\spal2
\begin{vers}
\textit{"Hun"}:
Du skiv'r som lynet, 
hver gang jeg kigger over
Har mistet synet, 
koden den skiver for sig
Hvad skal jeg gøre, 
når jeg kon-fi-gu-rer'
Mit tast er svundet, 
ctrl det har jeg midstet
\textit{"Han"}:
\textbf{Dit sprog er simpelt, 
Python det kan dog noget
Jeg er jo single, 
så der er masser' af tid
Hvad kan jeg gøre, 
hvad har jeg af projekter?}
\textit{"Hun"}:
Du har så mange, 
dog ved jeg ikke helt for
\end{vers}
\begin{omkvaed}
\textbf{Hvad skal vi kode i nat?}
Nu har jeg drikket monner
\textbf{Hvad skal vi kode i nat?}
Det blir' sku' Stregsystemet
\textbf{Hvis du vil lave commits}
La' os få det på det rene,
de er såmænd pæne
Dog ka' de ikke køre
\textbf{Hvad skal vi kode i nat?}
Formater din kode
\textbf{Hvad skal vi kode i nat?}
Lad vær' at spørge mig (red: me-ig)
\textbf{Hvis du vil lave commits}
Så må du føle den chance,
når du i kadance
Med din kompilering
\end{omkvaed}
\vbox{}\vfill
\begin{vers}
\textit{"Hun"}:
Det er utroligt, 
hvilke sprog der fines
Kender du COBOl, 
det er sku ik så simpel
Mine kære Nim lang, 
gid der var nogn' der brugt' dig
Vi skiver i C Sharp, 
når vi er produktive
\textit{"Han"}:
\textbf{Jeg bruger mit eget - 
sprog, når det kompilerer
Nu har jeg tvunget, 
det til at ignorer
Alle min' stav'fejl, 
ja, den gætter bare
Og hvad der så sker? 
Ja, det ik' til at svare}
\end{vers}
\begin{omkvaed}
\textbf{Hvad skal vi kode i nat?}
Jeg tog mig ogs' en porter
\textbf{Hvad skal vi kode i nat?}
Sangbogen, ja så går det
\textbf{Hvis du vil lave commits}
Typst er overlegne,
\LaTeX{} gir' migræne
Når man skiver i det
\textbf{Hvad skal vi kode i nat?}
Jeg har et miniprojekt
\textbf{Hvad skal vi kode i nat?}
Det blev et større projekt
\textbf{Hvis du vil lave commits}
Ja, jeg kan gi' dig en chance,
når du mister balance
Ved at se min kode
\end{omkvaed}
\laps
\end{sang}
